## 백준 문제 풀이
- 문제 : https://www.acmicpc.net/problem/1546
- 코드

```C++
#include <iostream>
#include <vector>

using namespace std;

int main() {
	int n, max = 0, score;
	float sum = 0;
	vector<int> scores;

	cin >> n;
	for (int i = 0; i < n; i++)
	{
		cin >> score;
		scores.push_back(score);
		if (scores[i] > max) {
			max = scores[i];
		}
	}
	
	for (int i = 0; i < n; i++) {
		sum += (float)scores[i] / (float)max * 100;
	}
	cout << sum / (float)n << endl;
	return 0;
	
}
```
