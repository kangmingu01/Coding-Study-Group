## 백준 문제풀이(C++)

-----
- 문제 : https://www.acmicpc.net/problem/2292
- 코드

```C++
#include <iostream>

using namespace std;

int main() {
	int n, cnt = 1, num = 0;
	cin >> n;

	if (n == 1) { cout << cnt << endl; }
	else {
		int prev = 2;
		for (int i = 0; i < n; i++) {
			num = prev + 6*i;
			if (n < num) {
				cout << cnt << endl;
				break;
			}
			else {
				prev = num;
				cnt++;
			}
		}
	}
	return 0;
}
```
