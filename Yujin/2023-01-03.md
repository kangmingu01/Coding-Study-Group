## 백준 단계별문제풀이(C++)

-----
- 문제 : https://www.acmicpc.net/problem/2480
- 코드

```C++
#include <iostream>
using namespace std;

int main()
{
	int a, b, c;
	cin >> a >> b >> c;
	if (a == b && a == c)
	{
		cout << 10000 + a * 1000 << endl;
	}
	else if (a == b || a == c)
	{
		cout << 1000 + a * 100 << endl;
	}
	else if (b == c)
	{
		cout << 1000 + b * 100 << endl;
	}
	else
	{
		cout << max(a, max(b, c)) * 100 << endl;
	}
}
```

-----
- 문재 : https://www.acmicpc.net/problem/2884
- 코드

```C++
#include <iostream>
using namespace std;

int main() {
	int h, m;
	cin >> h >> m;
	if (m - 45 < 0) {
		if (h == 0) {
			cout << 23 << " " << m + 15 << endl;
		}
		else {
			cout << h - 1 << " " << m + 15 << endl;
		}
	}
	else {
		cout << h << " " << m - 45 << endl;
	}
	return 0;
}
```
